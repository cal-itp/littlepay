38 def _format_expiry(self, expiry: datetime): 

39 """Formats an expiry datetime into a string suitable for using in an API request body.""" 

40 if not isinstance(expiry, datetime): 

41 raise TypeError("expiry must be a Python datetime instance") 

42 # determine if expiry is an "aware" or "naive" datetime instance 

43 # https://docs.python.org/3/library/datetime.html#determining-if-an-object-is-aware-or-naive 

44 if expiry.tzinfo is not None and expiry.tzinfo.utcoffset(expiry) is not None: 

45 # expiry is an "aware" datetime instance, meaning it has associated time zone information 

46 # ensure this datetime instance is expressed in UTC 

47 expiry = expiry.astimezone(timezone.utc) 

48 else: 

49 # expiry is a "naive" datetime instance, meaning it has no associated time zone information 

50 # assume this datetime instance was provided in UTC 

51 expiry = expiry.replace(tzinfo=timezone.utc) 

52 # now expiry is an "aware" datetime instance in UTC, format and add to the payload 

53 # datetime.isoformat() adds the UTC offset like +00:00 

54 # so keep everything but the last 6 characters and add the Z offset character 

55 return f"{expiry.isoformat(timespec='seconds')[:-6]}Z"
